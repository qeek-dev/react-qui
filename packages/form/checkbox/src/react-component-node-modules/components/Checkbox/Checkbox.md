### Light Theme

```js
initialState = { checked: false }
;<div>
  <div>
    <Checkbox
      id="normal"
      value="normal"
      checked={state.checked}
      onClick={e => {
        console.log(`checked: ${e.target.checked}, value: ${e.target.value}`)
        setState({ checked: !state.checked })
      }}
    >
      test
    </Checkbox>
  </div>
  <div>
    <Checkbox
      id="disabled"
      value="disabled"
      disabled
      checked={false}
      onClick={e => setState({ checked: !state.checked })}
    >
      test
    </Checkbox>
  </div>
  <div>
    <Checkbox
      id="disabled-checked"
      value="disabled-checked"
      disabled
      checked={true}
      onClick={e => setState({ checked: !state.checked })}
    >
      test
    </Checkbox>
  </div>
</div>
```
